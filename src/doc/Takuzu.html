def Takuzu.launch()
        Gtk.init
        @builder = Gtk::Window.new()
        @builder.set_default_size(320,240)
        @builder.set_title("Takuzu Deluxe - " + $joueur.pseudo)
        
        langue = Langue.new('fr')
        Menu.afficher(@builder, langue)

        @builder.show_all
        Gtk.main
end
